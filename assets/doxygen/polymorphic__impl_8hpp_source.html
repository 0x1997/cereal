185  template <class Archive, class T> struct OutputBindingCreator
186  {
188  static void writeMetadata(Archive & ar)
189  {
190  // Register the polymorphic type name with the archive, and get the id
191  char const * name = binding_name<T>::name();
192  std::uint32_t id = ar.registerPolymorphicType(name);
193 
194  // Serialize the id
195  ar( _CEREAL_NVP("polymorphic_id", id) );
196 
197  // If the msb of the id is 1, then the type name is new, and we should serialize it
198  if( id & detail::msb_32bit )
199  {
200  std::string namestring(name);
201  ar( _CEREAL_NVP("polymorphic_name", namestring) );
202  }
