190  std::uint32_t registerSharedPointer( void const * addr )
191  {
192  // Handle null pointers by just returning 0
193  if(addr == 0) return 0;
194 
195  auto id = itsSharedPointerMap.find( addr );
196  if( id == itsSharedPointerMap.end() )
197  {
198  auto ptrId = itsCurrentPointerId++;
199  itsSharedPointerMap.insert( {addr, ptrId} );
200  return ptrId | detail::msb_32bit; // mask MSB to be 1
201  }
202  else
203  return id->second;
204  }
