229  void writeName()
230  {
231  NodeType const & nodeType = itsNodeStack.top();
232 
233  // Start up either an object or an array, depending on state
234  if(nodeType == NodeType::StartArray)
235  {
236  itsWriter.StartArray();
237  itsNodeStack.top() = NodeType::InArray;
238  }
239  else if(nodeType == NodeType::StartObject)
240  {
241  itsNodeStack.top() = NodeType::InObject;
242  itsWriter.StartObject();
243  }
244 
245  // Array types do not output names
246  if(nodeType == NodeType::InArray) return;
247 
248  if(itsNextName == nullptr)
249  {
250  std::string name = "value" + std::to_string( itsNameCounter.top()++ ) + "\0";
251  saveValue(name);
252  }
253  else
254  {
255  saveValue(itsNextName);
256  itsNextName = nullptr;
257  }
258  }

439  inline void search()
440  {
441  // The name an NVP provided with setNextName()
442  if( itsNextName )
443  {
444  // The actual name of the current node
445  auto const actualName = itsIteratorStack.back().name();
446 
447  // Do a search if we don't see a name coming up, or if the names don't match
448  if( !actualName || std::strcmp( itsNextName, actualName ) != 0 )
449  itsIteratorStack.back().search( itsNextName );
450  }
451 
452  itsNextName = nullptr;
453  }
