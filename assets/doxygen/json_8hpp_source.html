317  void writeName()
318  {
319  NodeType const & nodeType = itsNodeStack.top();
320 
321  // Start up either an object or an array, depending on state
322  if(nodeType == NodeType::StartArray)
323  {
324  itsWriter.StartArray();
325  itsNodeStack.top() = NodeType::InArray;
326  }
327  else if(nodeType == NodeType::StartObject)
328  {
329  itsNodeStack.top() = NodeType::InObject;
330  itsWriter.StartObject();
331  }
332 
333  // Array types do not output names
334  if(nodeType == NodeType::InArray) return;
335 
336  if(itsNextName == nullptr)
337  {
338  std::string name = "value" + std::to_string( itsNameCounter.top()++ ) + "\0";
339  saveValue(name);
340  }
341  else
342  {
343  saveValue(itsNextName);
344  itsNextName = nullptr;
345  }
346  }
347 
349  void makeArray()
350  {
351  itsNodeStack.top() = NodeType::StartArray;
352  }

503  inline void search( const char * searchName )
504  {
505  const auto len = std::strlen( searchName );
506  size_t index = 0;
507  for( auto it = itsMemberItBegin; it != itsMemberItEnd; ++it, ++index )
508  {
509  const auto currentName = it->name.GetString();
510  if( ( std::strncmp( searchName, currentName, len ) == 0 ) &&
511  ( std::strlen( currentName ) == len ) )
512  {
513  itsIndex = index;
514  return;
515  }
516  }
517 
518  throw Exception("JSON Parsing failed - provided NVP (" + std::string(searchName) + ") not found");
519  }

522  MemberIterator itsMemberItBegin, itsMemberItEnd;
523  ValueIterator itsValueItBegin, itsValueItEnd;
524  size_t itsIndex;
525  enum Type {Value, Member, Null_} itsType;
526  };

537  inline void search()
538  {
539  // The name an NVP provided with setNextName()
540  if( itsNextName )
541  {

544 
545  // Do a search if we don't see a name coming up, or if the names don't match
546  if( !actualName || std::strcmp( itsNextName, actualName ) != 0 )
547  itsIteratorStack.back().search( itsNextName );
