308  void writeName()
309  {
310  NodeType const & nodeType = itsNodeStack.top();
311 
312  // Start up either an object or an array, depending on state
313  if(nodeType == NodeType::StartArray)
314  {
315  itsWriter.StartArray();
316  itsNodeStack.top() = NodeType::InArray;
317  }
318  else if(nodeType == NodeType::StartObject)
319  {
320  itsNodeStack.top() = NodeType::InObject;
321  itsWriter.StartObject();
322  }
323 
324  // Array types do not output names
325  if(nodeType == NodeType::InArray) return;
326 
327  if(itsNextName == nullptr)
328  {
329  std::string name = "value" + std::to_string( itsNameCounter.top()++ ) + "\0";
330  saveValue(name);
331  }
332  else
333  {
334  saveValue(itsNextName);
335  itsNextName = nullptr;
336  }
337  }
338 
340  void makeArray()
341  {
342  itsNodeStack.top() = NodeType::StartArray;
343  }

489  {

491  size_t index = 0;

493  if( std::strncmp( searchName, it->name.GetString(), len ) == 0 )
